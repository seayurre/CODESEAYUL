#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <algorithm>
#pragma warning(disable:4996)
using namespace std;
typedef struct onezero
{
 long long int onecount;
 long long int zerocount;
}O;
int main()
{
 int t;
 scanf("%d",&t);
 int n[1000];
 for (int i = 0; i < t; i++)
 {
  scanf("%d", &n[i]);
 }
 O a[41];
 a[0].zerocount = 1;
 a[0].onecount = 0;
 a[1].zerocount = 0;
 a[1].onecount = 1;
 for (int i = 2; i <= 40; i++)
 {
  a[i].onecount = a[i - 1].onecount + a[i - 2].onecount;
  a[i].zerocount = a[i - 1].zerocount + a[i - 2].zerocount;
 }
 
 for (int i = 0; i < t; i++)
 {
  printf("%lld %lld", a[n[i]].zerocount, a[n[i]].onecount);
  printf("\n");
 }
}

*편의상 문제의 fibonacci()를 f()로 표기
간단히 설명하자면, 피보나치 함수이므로 f(n)이 실행되면 f(n-1)과 f(n-2)가 호출되게 됩니다.
구조체를 이용해 각 f(n)의 0의 개수와 1의 개수를 저장하고, f(n)을 구할 때는 f(n-1)의 값과 f(n-2)의 값을 더합니다.
이런식으로 f(n)의 0의 개수와 1의 개수를 구해가다가 출력하면 끝!


또는 다이나믹 프로그래밍(dp)을 이용해 조금 더 개념과 코드를 간결하게 만들 수도 있습니다.

#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <cstring>
#pragma warning(disable:4996)
using namespace std;
 
int dp[50][2];
 
int main()
{
    int t;
    scanf("%d", &t);
    for (int ts = 1; ts <= t; ts++)
    {
        int n;
        scanf("%d", &n);
 
        dp[0][0] = 1;
        dp[1][1] = 1;
        for (int i = 2; i <= n; i++)
        {
            dp[i][0] = dp[i - 1][0] + dp[i - 2][0];
            dp[i][1] = dp[i - 1][1] + dp[i - 2][1];
        }
 
        printf("%d %d\n", dp[n][0], dp[n][1]);
        memset(dp, 0, sizeof(dp));
    }
}
